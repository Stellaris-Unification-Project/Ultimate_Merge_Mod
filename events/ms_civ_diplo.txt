namespace = ms_diplo;

country_event = {
	id = ms_diplo.1
	
	title = "Greetings"
	
	desc = {
		text = ms_diplo.1.desca
		trigger = {
			from = {
				NOT = {
					has_country_flag = ms_energy_traded
					has_country_flag = ms_minerals_traded
				}
			}
		}
	}
	
	desc = {
		text = ms_diplo.1.descb
		trigger = {
			from = {
				has_country_flag = ms_energy_traded
				has_country_flag = ms_minerals_traded
			}
		}
	}
	
	diplomatic = yes
	
	picture_event_data = {
		portrait = from
		planet_background = from
		room = root
	}
	
	is_triggered_only = yes
	
	trigger = {
		FROM = {
			is_country_type = ms_post_primitive
			capital_scope = {
				solar_system = {
					space_owner = {
						is_same_value = root
					}
				}
			}
		}
	}
	
	option = {
		name = ms_diplo.1.c
		response_text = ms_diplo.1.c.response
		
	}
	
	option = {
		name = ms_diplo.1.a
		response_text = ms_diplo.1.a.response
		trigger = {
			from = {
				NOT = {
					has_country_flag = ms_minerals_traded
				}
				NOT = {
					any_owned_pop = {
						is_robot_pop = yes
					}
				}
			}
		}
		allow = { 
			energy > 1500
		}
		add_minerals = 1000
		add_energy = -1500
		hidden_effect = {
			from = {
				set_timed_country_flag = {
					flag = ms_minerals_traded
					days = 1800
				}
			}
		}
	}
	
	option = {
		name = ms_diplo.1.d
		custom_tooltip = ms_diplo.1.d.tooltip
		response_text = ms_diplo.1.d.response
		trigger = {
			from = {
				NOT = {
					has_country_flag = ms_minerals_traded
				}
				capital_scope = {
					any_owned_pop = {
						is_robot_pop = yes
					}
				}
			}
		}
		allow = { 
			energy > 1500
		}
		add_minerals = 2000
		add_energy = -1500
		hidden_effect = {
			from = {
				set_timed_country_flag = {
					flag = ms_minerals_traded
					days = 1800
				}
			}
		}
	}
	
	option = {
		name = ms_diplo.1.b
		response_text = ms_diplo.1.b.response
		trigger = {
			from = {
				NOT = {
					has_country_flag = ms_energy_traded
				}
			}
		}
		allow = { 
			minerals > 1500
		}
		add_energy = 1000
		add_minerals = -1500
		hidden_effect = {
			from = {
				set_timed_country_flag = {
					flag = ms_energy_traded
					days = 1800
				}
			}
		}
	}	
}
