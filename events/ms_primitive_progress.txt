namespace = ms_prim_prog;

country_event = {
	id = ms_prim_prog.1
	hide_window = yes

	trigger = {
		is_country_type = ms_primitive_1
		has_country_flag = stone_age
		NOT = { has_country_flag = ms_advance_denied }
		NOT = { has_country_flag = recently_advanced }
	}

	mean_time_to_happen = {
		years = 160
		modifier = {
			factor = 0.1
			has_country_flag = ms_advance_accelerated
		}
	}

	immediate = {
		set_timed_country_flag = {
			flag = recently_advanced
			days = 100
		}
		remove_country_flag = stone_age
		set_country_flag = bronze_age
		set_primitive_age = bronze_age
		capital_scope = {
			random_tile = {
				limit = { 
					has_blocker = no 
					has_building = no 
					has_grown_pop = yes
				}
				set_building = "building_primitive_farm"
			}
			random_tile = {
				limit = { 
					has_blocker = no 
					has_building = no 
					has_grown_pop = yes
				}
				set_building = "building_primitive_farm"
			}			
			space_owner = {
				country_event = { id = ms_prim_prog.2 }
			}
		}
	}
}

country_event = {
	id = ms_prim_prog.11
	hide_window = yes

	trigger = {
		is_country_type = ms_primitive_1
		has_country_flag = steam_age
		NOT = { has_country_flag = ms_advance_denied }
		NOT = { has_country_flag = recently_advanced }
	}

	mean_time_to_happen = {
		years = 100
		modifier = {
			factor = 0.1
			has_country_flag = ms_advance_accelerated
		}
	}

	immediate = {
		set_timed_country_flag = {
			flag = recently_advanced
			days = 100
		}
		remove_country_flag = steam_age
		change_government = {
			civics = {
				civic = civic_increasing_urbanization
				civic = civic_atmospheric_pollution
			}
		}
		set_country_flag = industrial_age
		set_primitive_age = industrial_age
		set_graphical_culture = industrial_01
		capital_scope = {
			create_army = {
				name = "Industrial Army"
				owner = root
				species = root
				type = "industrial_army"
			}
			every_tile = {
				limit = {
					has_blocker = yes
				}
				remove_blocker = yes
			}
			while = {
				count = 3
				random_tile = {
					limit = {
						has_blocker = no
						has_building = no
						has_grown_pop = no
					}
					set_blocker = tb_decrepit_dwellings
				}
				random_tile = {
					limit = {
						has_blocker = no
						has_building = no
						has_grown_pop = no
					}
					set_blocker = tb_failing_infrastructure
				}
			}
			random_tile = {
				limit = {
					has_blocker = no
					has_building = no
					has_grown_pop = no
				}
				create_pop = {
					species = owner
				}
				clear_deposits = yes
				add_deposit = d_mineral_deposit
				set_building = "building_primitive_factory"
			}
			space_owner = {
				country_event = { id = ms_prim_prog.12 }
			}
		}
	}
}
