country_event = {
	id = ms_rand_ev.1
	hide_window = yes
	
	trigger = {
		is_country_type = ms_primitive_1

			has_country_flag = abnegation_age

			has_country_flag = retaliation_age
		}
		NOT = {
			has_country_flag = ms_rand_ev_1
		}
	}
	
	mean_time_to_happen = {
		months = 240
	}
	
	immediate = {
		set_timed_country_flag = {

		}

			solar_system = {
				space_owner = {

						id = ms_rand_ev.2
					}
				}
			}
		}
	}
}
